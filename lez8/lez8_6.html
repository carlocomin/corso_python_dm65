# 1) Conteggio Partizioni
from functools import lru_cache

@lru_cache(maxsize=None)
def partizioni(n, max_somma):
    if n == 0:
        return 1
    if n < 0 or max_somma == 0:
        return 0
    return partizioni(n - max_somma, max_somma) + partizioni(n, max_somma - 1)

# 2) Cammini in Griglia NxM
@lru_cache(maxsize=None)
def cammini(n, m):
    if n == 0 or m == 0:
        return 1
    return cammini(n-1, m) + cammini(n, m-1)

# 3) Minimo Numero di Monete
def min_monete(valore, monete):
    memo = {}
    def dp(v):
        if v == 0:
            return 0
        if v in memo:
            return memo[v]
        res = float('inf')
        for coin in monete:
            if v >= coin:
                res = min(res, 1 + dp(v - coin))
        memo[v] = res
        return res
    return dp(valore)

# 4) Longest Common Subsequence (LCS)
@lru_cache(maxsize=None)
def lcs(s1, s2):
    if not s1 or not s2:
        return 0
    if s1[0] == s2[0]:
        return 1 + lcs(s1[1:], s2[1:])
    return max(lcs(s1[1:], s2), lcs(s1, s2[1:]))

# 5) Palindrome Partitioning
@lru_cache(maxsize=None)
def min_cut(s):
    if s == s[::-1]:
        return 0
    res = len(s)-1
    for i in range(1, len(s)):
        if s[:i] == s[:i][::-1]:
            res = min(res, 1 + min_cut(s[i:]))
    return res
