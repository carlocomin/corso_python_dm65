def merge_sort(lista):
    if len(lista) <= 1:
        return lista

    centro = len(lista) // 2
    sx = merge_sort(lista[:centro])
    dx = merge_sort(lista[centro:])

    return merge(sx, dx)

def merge(sx, dx):
    risultato = []
    i, j = 0, 0
    while i < len(sx) and j < len(dx):
        if sx[i] < dx[j]:
            risultato.append(sx[i])
            i += 1
        else:
            risultato.append(dx[j])
            j += 1
    risultato += sx[i:]
    risultato += dx[j:]
    return risultato
