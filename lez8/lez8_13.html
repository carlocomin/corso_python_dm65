def quick_sort(lista):
    if len(lista) <= 1:
        return lista

    pivot = lista[0]
    minori = [x for x in lista[1:] if x < pivot]
    maggiori = [x for x in lista[1:] if x >= pivot]

    return quick_sort(minori) + [pivot] + quick_sort(maggiori)
