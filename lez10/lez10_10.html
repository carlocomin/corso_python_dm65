import heapq

def dijkstra(grafo, sorgente):
    distanza = {nodo: float('inf') for nodo in grafo}
    distanza[sorgente] = 0
    coda = [(0, sorgente)]  # (distanza, nodo)

    while coda:
        dist_corrente, nodo_corrente = heapq.heappop(coda)
        if dist_corrente > distanza[nodo_corrente]:
            continue
        for vicino, peso in grafo[nodo_corrente]:
            nuova_dist = dist_corrente + peso
            if nuova_dist < distanza[vicino]:
                distanza[vicino] = nuova_dist
                heapq.heappush(coda, (nuova_dist, vicino))

    return distanza

# Esempio di grafo (lista adiacenza con pesi)
grafo = {
    0: [(1, 4), (2, 5)],
    1: [(2, 1), (3, 2)],
    2: [(3, 1)],
    3: []
}

risultato = dijkstra(grafo, 0)
print("Distanze minime dalla sorgente:", risultato)
