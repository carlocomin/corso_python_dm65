# 2. Grafo da input con Dijkstra
def costruisci_grafo_da_input():
    n = int(input("Numero di nodi: "))
    e = int(input("Numero di archi: "))
    grafo = {i: [] for i in range(n)}
    for _ in range(e):
        u, v, p = map(int, input("Arco u v peso: ").split())
        grafo[u].append((v, p))
    return grafo
