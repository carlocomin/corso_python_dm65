def dfs(grafo, nodo, visitati=None):
    if visitati is None:
        visitati = set()

    print(nodo, end=" ")
    visitati.add(nodo)

    for vicino in grafo[nodo]:
        if vicino not in visitati:
            dfs(grafo, vicino, visitati)

# Esempio di grafo
grafo = {
    0: [1, 2],
    1: [0, 3],
    2: [0, 4],
    3: [1],
    4: [2]
}

dfs(grafo, 0)  # Output: 0 1 3 2 4
