def bellman_ford(grafo, V, sorgente):
    distanza = [float('inf')] * V
    distanza[sorgente] = 0

    # Rilassa tutti gli archi V-1 volte
    for _ in range(V - 1):
        for (u, v, peso) in grafo:
            if distanza[u] != float('inf') and distanza[u] + peso < distanza[v]:
                distanza[v] = distanza[u] + peso

    # Controlla cicli negativi
    for (u, v, peso) in grafo:
        if distanza[u] != float('inf') and distanza[u] + peso < distanza[v]:
            print("Rilevato ciclo negativo!")
            return None

    return distanza

# Esempio di grafo (lista di archi: u, v, peso)
grafo = [
    (0, 1, 4),
    (0, 2, 5),
    (1, 2, -3),
    (2, 3, 4),
    (3, 1, 6)
]

V = 4  # Numero di nodi
sorgente = 0

risultato = bellman_ford(grafo, V, sorgente)
if risultato:
    print("Distanze minime dalla sorgente:", risultato)
