from collections import deque

def bfs(grafo, partenza):
    visitati = set()
    coda = deque([partenza])

    while coda:
        nodo = coda.popleft()
        if nodo not in visitati:
            print(nodo, end=" ")
            visitati.add(nodo)
            for vicino in grafo[nodo]:
                if vicino not in visitati:
                    coda.append(vicino)

# Esempio di grafo
grafo = {
    0: [1, 2],
    1: [0, 3],
    2: [0, 4],
    3: [1],
    4: [2]
}

bfs(grafo, 0)  # Output: 0 1 2 3 4
