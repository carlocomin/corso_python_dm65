class Libro:
    def __init__(self, titolo, autore):
        self.titolo = titolo
        self.autore = autore
        self.in_prestito = False

    def __str__(self):
        stato = "In prestito" if self.in_prestito else "Disponibile"
        return f"{self.titolo} di {self.autore} â€“ {stato}"

class Biblioteca:
    def __init__(self):
        self.catalogo = []

    def aggiungi_libro(self, libro):
        self.catalogo.append(libro)

    def stampa_libri(self):
        for libro in self.catalogo:
            print(libro)

    def presta_libro(self, titolo):
        for libro in self.catalogo:
            if libro.titolo == titolo and not libro.in_prestito:
                libro.in_prestito = True
                print(f"Hai preso in prestito: {titolo}")
                return
        print("Libro non disponibile.")

    def restituisci_libro(self, titolo):
        for libro in self.catalogo:
            if libro.titolo == titolo and libro.in_prestito:
                libro.in_prestito = False
                print(f"Hai restituito: {titolo}")
                return
        print("Restituzione non valida.")
