class Giocatore:
    def __init__(self, nome):
        self.nome = nome
        self.lucidita = 100         # Punteggio mentale (100 = stabile, 0 = psicosi)
        self.inventario = []        # Lista di oggetti raccolti
        self.storia_scelte = []     # Elenco delle scene attraversate

    def aggiorna_lucidita(self, delta):
        self.lucidita += delta
        self.lucidita = max(0, min(self.lucidita, 100))  # Clamp tra 0 e 100

    def aggiungi_oggetto(self, oggetto):
        if oggetto not in self.inventario:
            self.inventario.append(oggetto)

    def registra_scelta(self, scena):
        self.storia_scelte.append(scena)
