  4. Indovina il Numero 
    
import random
numero_segreto = random.randint(1, 100)
while True:
    tentativo = int(input("Indovina il numero: "))
    if tentativo > numero_segreto:
        print("Troppo alto!")
    elif tentativo < numero_segreto:
        print("Troppo basso!")
    else:
        print("Indovinato!")
        break
