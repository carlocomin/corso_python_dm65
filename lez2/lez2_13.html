  3. Trova il Massimo Comune Divisore (MCD) 
    
def mcd(a, b):
    while b:
        a, b = b, a % b
    return a

x, y = map(int, input("Inserisci due numeri: ").split())
print("MCD:", mcd(x, y))

  7. Numeri Amici 
    
def somma_divisori(n):
    return sum([i for i in range(1, n) if n % i == 0])

a, b = map(int, input("Inserisci due numeri: ").split())
if somma_divisori(a) == b and somma_divisori(b) == a:
    print("Sono numeri amici")
else:
    print("Non sono numeri amici")
  8. Crivello di Eratostene 
    
def crivello_eratostene(n):
    primi = [True] * (n + 1)
    p = 2
    while p ** 2 <= n:
        if primi[p]:
            for i in range(p * p, n + 1, p):
                primi[i] = False
        p += 1
    return [p for p in range(2, n + 1) if primi[p]]

N = int(input("Trova i numeri primi fino a: "))
print("Numeri primi:", crivello_eratostene(N))
  9. Somma delle Cifre Pari e Dispari 
    
n = input("Inserisci un numero: ")
somma_pari = sum(int(c) for c in n if int(c) % 2 == 0)
somma_dispari = sum(int(c) for c in n if int(c) % 2 != 0)
print(f"Somma cifre pari: {somma_pari}, Somma cifre dispari: {somma_dispari}")
