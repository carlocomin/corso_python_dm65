def factorial(n):
    risultato = 1
    for i in range(2, n + 1):
        risultato *= i
    return risultato

def conta_lettere(s):
    d = {}
    for c in s:
        if c.isalpha():
            d[c] = d.get(c, 0) + 1
    return d

def filtra_pari(lista):
    return list(filter(lambda x: x % 2 == 0, lista))
