# 1. Somma Massima Sottosequenza (Kadane)
def max_subarray(nums):
    max_corrente = max_totale = nums[0]
    for num in nums[1:]:
        max_corrente = max(num, max_corrente + num)
        max_totale = max(max_totale, max_corrente)
    return max_totale

# 2. Ricerca Binaria
def binary_search(arr, target):
    sx, dx = 0, len(arr) - 1
    while sx <= dx:
        mid = (sx + dx) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            sx = mid + 1
        else:
            dx = mid - 1
    return -1

# 3. Parentesi Bilanciate
def parentesi_bilanciate(s):
    stack = []
    mappa = {')':'(', '}':'{', ']':'['}
    for c in s:
        if c in mappa.values():
            stack.append(c)
        elif c in mappa.keys():
            if not stack or mappa[c] != stack.pop():
                return False
    return not stack

# 4. Frequenza Numeri
from collections import Counter
def frequenza(arr):
    return dict(Counter(arr))
