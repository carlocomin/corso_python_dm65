# 5. Minimo Numero di Salti
def salti_minimi(nums):
    salti = 0
    fine = 0
    massimo = 0
    for i in range(len(nums) - 1):
        massimo = max(massimo, i + nums[i])
        if i == fine:
            salti += 1
            fine = massimo
    return salti

# 6. Sottosequenza Crescente Massima (LIS)
def LIS(nums):
    dp = []
    import bisect
    for num in nums:
        i = bisect.bisect_left(dp, num)
        if i == len(dp):
            dp.append(num)
        else:
            dp[i] = num
    return len(dp)

# 7. Verifica Grafo Bipartito
def bipartito(n, edges):
    grafo = {i: [] for i in range(n)}
    for u, v in edges:
        grafo[u].append(v)
        grafo[v].append(u)
    colore = {}
    from collections import deque
    for nodo in range(n):
        if nodo not in colore:
            queue = deque([nodo])
            colore[nodo] = 0
            while queue:
                u = queue.popleft()
                for v in grafo[u]:
                    if v not in colore:
                        colore[v] = 1 - colore[u]
                        queue.append(v)
                    elif colore[v] == colore[u]:
                        return False
    return True

# 8. Merge Intervalli
def merge_intervalli(intervalli):
    intervalli.sort(key=lambda x: x[0])
    merged = []
    for interv in intervalli:
        if not merged or merged[-1][1] < interv[0]:
            merged.append(interv)
        else:
            merged[-1][1] = max(merged[-1][1], interv[1])
    return merged
