import pygame
LARGHEZZA, ALTEZZA = 600, 400
righe, colonne = 20, 30
TILE = LARGHEZZA // colonne

# Colori
BIANCO = (255,255,255)
NERO = (0,0,0)
VERDE = (0,255,0)
ROSSO = (255,0,0)
GRIGIO = (50,50,50)

pygame.init()
schermo = pygame.display.set_mode((LARGHEZZA, ALTEZZA))
pygame.display.set_caption("A* - Setup Griglia")

üîπ Funzione di disegno

def disegna_griglia(schermo, griglia, start, goal):
    for riga in griglia:
        for nodo in riga:
            x, y = nodo.x * TILE, nodo.y * TILE
            colore = BIANCO
            if nodo.ostacolo:
                colore = NERO
            elif nodo == start:
                colore = VERDE
            elif nodo == goal:
                colore = ROSSO
            pygame.draw.rect(schermo, colore, (x, y, TILE-1, TILE-1))

# Esempio di uso
disegna_griglia(schermo, griglia, start, goal)
