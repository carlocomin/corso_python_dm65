def disegna_griglia(griglia, start, goal):
    for riga in griglia:
        for nodo in riga:
            x = nodo.x * DIM_CELLA
            y = nodo.y * DIM_CELLA
            colore = (255, 255, 255)
            if nodo.ostacolo:
                colore = (0, 0, 0)
            elif nodo == start:
                colore = (0, 255, 0)
            elif nodo == goal:
                colore = (255, 0, 0)
            pygame.draw.rect(screen, colore, (x, y, DIM_CELLA, DIM_CELLA))
            pygame.draw.rect(screen, (200, 200, 200), (x, y, DIM_CELLA, DIM_CELLA), 1)
