Lezione 1: Esempi di Codice - Variabili, Operatori, Tipi di Dati

ðŸ”¹ Dichiarazione e Assegnazione di Variabili


# Assegnazione di variabili
numero = 42
pi_greco = 3.14
saluto = "Ciao, mondo!"
e_vero = True

# Stampa delle variabili
print(numero)  # Output: 42
print(pi_greco)  # Output: 3.14
print(saluto)  # Output: Ciao, mondo!
print(e_vero)  # Output: True

ðŸ”¹ Operatori Aritmetici in Azione


# Operazioni matematiche
x = 10
y = 3

somma = x + y
sottrazione = x - y
moltiplicazione = x * y
divisione = x / y  # Risultato float
modulo = x % y  # Resto della divisione
esponenziale = x ** y  # x elevato alla y

print("Somma:", somma)
print("Sottrazione:", sottrazione)
print("Moltiplicazione:", moltiplicazione)
print("Divisione:", divisione)
print("Modulo:", modulo)
print("Esponenziale:", esponenziale)

ðŸ”¹ Conversione tra Tipi di Dati


# Conversione di tipi di dati
numero_str = str(100)  # Da int a stringa
numero_int = int("42")  # Da stringa a int
numero_float = float("3.14")  # Da stringa a float

print(numero_str, type(numero_str))  # Output: '100' 
print(numero_int, type(numero_int))  # Output: 42 
print(numero_float, type(numero_float))  # Output: 3.14
