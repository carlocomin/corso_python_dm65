# 1) Classe Stack
class Stack:
    def __init__(self):
        self.dati = []

    def push(self, x):
        self.dati.append(x)

    def pop(self):
        return self.dati.pop()

    def peek(self):
        return self.dati[-1]

    def empty(self):
        return len(self.dati) == 0

# 2) Parentesi Bilanciate
def bilanciate(s):
    stack = []
    for c in s:
        if c == '(':
            stack.append(c)
        elif c == ')':
            if not stack:
                return False
            stack.pop()
    return len(stack) == 0

# 3) Inverso di una lista
def inversa(lista):
    stack = []
    for x in lista:
        stack.append(x)
    risultato = []
    while stack:
        risultato.append(stack.pop())
    return risultato

# 4) Undo Editor
def undo_editor(comandi):
    testo = []
    undo = []
    for comando in comandi:
        if comando == "UNDO":
            if testo:
                undo.append(testo.pop())
        else:
            testo.append(comando)
    return testo

# 5) Decimale in Binario
def decimale_binario(n):
    stack = []
    while n > 0:
        stack.append(n % 2)
        n //= 2
    binario = ""
    while stack:
        binario += str(stack.pop())
    return binario
