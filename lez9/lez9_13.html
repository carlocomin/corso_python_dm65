# 1) Conta frequenza parole
def conta_parole(testo):
    parole = testo.split()
    frequenze = {}
    for parola in parole:
        frequenze[parola] = frequenze.get(parola, 0) + 1
    return frequenze

# 2) Numeri distinti
def numeri_distinti(lista):
    visti = set()
    risultato = []
    for num in lista:
        if num not in visti:
            visti.add(num)
            risultato.append(num)
    return risultato

# 3) Anagrammi
def anagrammi(s1, s2):
    return sorted(s1) == sorted(s2)

# Variante con dict
def anagrammi_dict(s1, s2):
    from collections import Counter
    return Counter(s1) == Counter(s2)

# 4) Elimina duplicati e ordina
def elimina_duplicati(nomi):
    return sorted(set(nomi))

# 5) Intersezione e Unione di due liste
def operazioni_insiemi(lista1, lista2):
    a = set(lista1)
    b = set(lista2)
    return a & b, a | b
