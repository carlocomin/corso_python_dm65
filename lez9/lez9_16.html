class Nodo:
    def __init__(self, valore):
        self.valore = valore
        self.sinistro = None
        self.destro = None

# 1. Visite preorder, inorder, postorder
def preorder(n):
    if n:
        print(n.valore)
        preorder(n.sinistro)
        preorder(n.destro)

def inorder(n):
    if n:
        inorder(n.sinistro)
        print(n.valore)
        inorder(n.destro)

def postorder(n):
    if n:
        postorder(n.sinistro)
        postorder(n.destro)
        print(n.valore)

# 2. Conta nodi
def conta_nodi(n):
    if n is None:
        return 0
    return 1 + conta_nodi(n.sinistro) + conta_nodi(n.destro)

# 3. Altezza albero
def altezza(n):
    if n is None:
        return 0
    return 1 + max(altezza(n.sinistro), altezza(n.destro))

# 4. Cerca valore
def cerca_valore(n, val):
    if n is None:
        return False
    if n.valore == val:
        return True
    return cerca_valore(n.sinistro, val) or cerca_valore(n.destro, val)

# 5. Albero di esempio
root = Nodo(10)
root.sinistro = Nodo(5)
root.destro = Nodo(15)
root.sinistro.sinistro = Nodo(3)
root.destro.destro = Nodo(20)

inorder(root)  # Output: 3, 5, 10, 15, 20
