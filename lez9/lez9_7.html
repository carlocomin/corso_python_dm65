# 1) Classe Queue
from collections import deque

class Queue:
    def __init__(self):
        self.dati = deque()

    def enqueue(self, x):
        self.dati.append(x)

    def dequeue(self):
        return self.dati.popleft()

    def front(self):
        return self.dati[0]

    def empty(self):
        return len(self.dati) == 0

# 2) Coda Clienti in Banca
def banca(clienti):
    queue = deque(clienti)
    while queue:
        cliente = queue.popleft()
        print("Servito cliente:", cliente)

# 4) Token Circolare
def token_ring(processi, passi):
    queue = deque(processi)
    for _ in range(passi):
        processo = queue.popleft()
        print("Token a:", processo)
        queue.append(processo)

def gestisci_eventi(eventi):
