üîπ Una semplice rete neurale

import torch
import torch.nn as nn
import torch.nn.functional as F

class ReteSemplice(nn.Module):
    def __init__(self):
        super().__init__()
        self.fc1 = nn.Linear(2, 4)   # input da 2 ‚Üí 4 neuroni nascosti
        self.fc2 = nn.Linear(4, 1)   # output da 4 ‚Üí 1 neurone

    def forward(self, x):
        x = F.relu(self.fc1(x))      # attivazione ReLU
        x = torch.sigmoid(self.fc2(x)) # output tra 0 e 1
        return x
